This plot illustrates the depth-damage curve of flooding for Katie‚Äôs Seafood House in Galveston near the Pier 21 water gauge. The every foot of flooding at the restaurant the Y axis shows the percentage of damage the flooding will have. The incline of this curve is mostly steady however when flooding reaches around 10 feet above ground the restaurant is expected to experience a higher increase in damages for each extra foot of flooding. So for example, a flood depth at the restaurant of 1 foot would equivalate to 20% damage whereas 12 feet of flooding at the restaurant would likely result in 50% damage. The expected damage reaches 100% just under 30 feet of flooding at the restaurant.

function get_depth_damage_function(
    depth_train::Vector{<:T}, dmg_train::Vector{<:AbstractFloat}
) where {T<:Unitful.Length}

    # interpolate
    depth_ft = ustrip.(u"ft", depth_train)
    interp_fn = Interpolations.LinearInterpolation(
        depth_ft,
        dmg_train;
        extrapolation_bc=Interpolations.Flat(),
    )

    damage_fn = function (depth::T2) where {T2<:Unitful.Length}
        return interp_fn(ustrip.(u"ft", depth))
    end
    return damage_fn 
end
